// Captura parâmetros da URL const params = new URLSearchParams(window.location.search); const bairro = params.get('nome') || "Bairro"; const img = params.get('img') || "mapa.png"; document.getElementById('tituloBairro').textContent = bairro; document.getElementById('mapaBairro').src = img; const form = document.getElementById('personForm'); const tableBody = document.querySelector('#personTable tbody'); let registros = []; // Carregar registros do backend async function loadRegistros() { try { const res = await fetch('/api/pessoas'); if (!res.ok) throw new Error("Erro ao carregar registros"); const data = await res.json(); registros = data.filter(r => r.bairro === bairro); renderTable(); } catch (err) { console.error(err); alert("Não foi possível carregar os registros."); } } // Renderizar tabela function renderTable() { tableBody.innerHTML = ''; registros.forEach((item, index) => { const row = document.createElement('tr'); row.innerHTML = ` ${item.dirigente} ${item.saida} `; tableBody.appendChild(row); }); } // Adicionar registro form.addEventListener('submit', async (e) => { e.preventDefault(); const dirigente = document.getElementById('dirigente').value; const saida = document.getElementById('saida').value; await fetch('/api/pessoas', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ bairro, dirigente, saida }) }); form.reset(); loadRegistros(); }); // Editar registro (remove e re-adiciona) async function editRegistro(index) { const registro = registros[index]; document.getElementById('dirigente').value = registro.dirigente; document.getElementById('saida').value = registro.saida; await deleteRegistro(registro.id); } // Remover registro async function deleteRegistro(id) { await fetch(`/api/pessoas?id=${id}`, { method: 'DELETE' }); loadRegistros(); } // Compartilhar link da imagem async function shareLink() { const imgUrl = document.getElementById('mapaBairro').src; if (navigator.share) { try { await navigator.share({ title: "Mapa do bairro", text: "Confira este mapa!", url: imgUrl }); } catch (err) { console.error("Erro ao compartilhar link:", err); } } else { alert("Compartilhamento de link não suportado neste navegador."); } } // Baixar imagem para dispositivos móveis (usando Blob) async function downloadImage() { const imgUrl = document.getElementById('mapaBairro').src; try { const response = await fetch(imgUrl); if (!response.ok) throw new Error("Erro ao buscar imagem"); const blob = await response.blob(); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = "mapa-bairro.png"; document.body.appendChild(a); a.click(); document.body.removeChild(a); URL.revokeObjectURL(url); } catch (err) { console.error("Erro ao baixar imagem:", err); alert("Não foi possível baixar a imagem neste dispositivo."); } } // Inicializar loadRegistros();
